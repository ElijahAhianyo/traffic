```python
def rotate_marker(path: Path, degrees: float) -> Path:
    angle = np.radians(degrees)
    newpath = path.vertices @ np.array([[np.cos(angle), -np.sin(angle)], [np.sin(angle), np.cos(angle)]])
    return Path(newpath, path.codes)
```
